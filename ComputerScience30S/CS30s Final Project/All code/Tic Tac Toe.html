 67     /**
 68      * What happens when a user clicks on a square (in the code)
 69      *
 70      * @param row the row that the square is in
 71      * @param column the column that the square is in
 72      */
 73     private void clickMouse(int row, int column) {
 74         click(row, column);
 75         count++;
 76         if (count >= 5) {
 77             String win = checkWin();
 78             if (!win.equals("")) {
 79                 JOptionPane.showMessageDialog(null, win);
 80                 if (playAgain() == true) {
 81                     this.dispose();
 82                     count = 0;
 83                     TicTacToe ticTacToe = new TicTacToe();
 84                 } else {
 85                     JOptionPane.showMessageDialog(
 86                             null,
 87                             "Thanks for playing TicTacToe",
 88                             Globals.title,
 89                             JOptionPane.PLAIN_MESSAGE
 90                     );
 91                     System.exit(0);
 92                 }
 93             } else if (win.equals("") && count == 9) {
 94                 JOptionPane.showMessageDialog(null, "It's a tie!");
 95                 if (playAgain() == true) {
 96                     this.dispose();
 97                     count = 0;
 98                     TicTacToe ticTacToe = new TicTacToe();
 99                 } else {
100                     JOptionPane.showMessageDialog(
101                             null,
102                             "Thanks for playing TicTacToe",
103                             Globals.title,
104                             JOptionPane.PLAIN_MESSAGE
105                     );
106                     System.exit(0);
107                 }
108             }
109         }
110     }

121     /**
122      * Ask the user a yes and no question, in a confirm dialog box
123      *
124      * @param text the yes no question to ask in the dialog
125      * @return true (yes), false (no)
126      */
127     private static boolean yesNo(String text) {
128         int response = JOptionPane.showConfirmDialog(null, text, Globals.title,
129                 JOptionPane.YES_NO_OPTION, JOptionPane.QUESTION_MESSAGE);
130         if (response == JOptionPane.YES_OPTION) {
131             return true;
132         } else {
133             return false;
134         }
135     }

137     /**
138      * This is what happens when the user clicks on a label (user interface)
139      *
140      * @param row what row the label is in
141      * @param column what column the label is in
142      */
143     private void click(int row, int column) {
144         if (count % 2 == 0 && matrix[row][column].getText().equals("")) {
145             matrix[row][column].setText(Globals.Xl);
146             matrix[row][column].setBackground(Color.blue);
147         } else if (count % 2 != 0 && matrix[row][column].getText().equals("")) {
148             matrix[row][column].setText(Globals.Ol);
149             matrix[row][column].setBackground(Color.yellow);
150         } else {
151             JOptionPane.showMessageDialog(null, "You need to chose a different square");
152             count--;
153         }
154     }

156     /**
157      * Checks to see if someone won
158      *
159      * @return the player who won
160      */
161     private String checkWin() {
162         String text = "";
163         if (text.equals("")) {
164             text = checkWinsGoingDown();
165         }
166         if (text.equals("")) {
167             text = checkWinsGoingAcross();
168         }
169         if (text.equals("")) {
170             text = checkWinsGoingDiagonally();
171         }
172         return text;
173     }

175     /**
176      * Checks to see if a player won diagonally
177      *
178      * @return the player who won
179      */
180     private String checkWinsGoingDiagonally() {
181         String text = "";
182         // Loops through the diagonals
183         if (matrix[0][0].getText().equals(Globals.Xl)
184                 && matrix[1][1].getText().equals(Globals.Xl)
185                 && matrix[2][2].getText().equals(Globals.Xl)) {
186             text = DIAGONALX;
187             return text;
188         } else if (matrix[0][2].getText().equals(Globals.Xl)
189                 && matrix[1][1].getText().equals(Globals.Xl)
190                 && matrix[2][0].getText().equals(Globals.Xl)) {
191             text = DIAGONALX;
192             return text;
193         }
194         if (matrix[0][0].getText().equals(Globals.Ol)
195                 && matrix[1][1].getText().equals(Globals.Ol)
196                 && matrix[2][2].getText().equals(Globals.Ol)) {
197             text = DIAGONALO;
198             return text;
199         } else if (matrix[0][2].getText().equals(Globals.Ol)
200                 && matrix[1][1].getText().equals(Globals.Ol)
201                 && matrix[2][0].getText().equals(Globals.Ol)) {
202             text = DIAGONALO;
203             return text;
204         }
205         return text;
206     }
